function generateZenerProblem() { hideAllInputGroups(); const canvas = document.getElementById('circuitCanvas'); if (!canvas) return; const ctx = canvas.getContext('2d'); // Reset canvas transform and clear ctx.setTransform(1, 0, 0, 1, 0, 0); ctx.clearRect(0, 0, canvas.width, canvas.height); if (typeof resetAnswerBox === "function") resetAnswerBox(); currentMode = 'zener'; const solBox = document.getElementById('solText'); if (solBox) solBox.style.display = "none"; // --- 1. RANDOM VARIABLES (WITH 800mA LIMIT) --- let Vz, R1, Vin, totalCurrent_mA, RL, switchClosed; const RL_options = [100, 200, 250, 400, 500]; // Loop until we find a combination where current <= 800mA let validProblem = false; while (!validProblem) { Vz = Math.floor(Math.random() * 5) + 5; // 5V to 9V R1 = (Math.floor(Math.random() * 5) + 1) * 100; // 100, 200, 300, 400, 500 Ohms const currentSteps = Math.floor(Math.random() * 10) + 5; Vin = Vz + (currentSteps * (R1 / 10)); RL = RL_options[Math.floor(Math.random() * RL_options.length)]; switchClosed = Math.random() > 0.5; // Calculate potential regulation const V_unregulated = Vin * (RL / (R1 + RL)); const isRegulating = V_unregulated > Vz; const finalVL = isRegulating ? Vz : V_unregulated; // Check current totalCurrent_mA = Math.round(((Vin - finalVL) / R1) * 1000); if (totalCurrent_mA <= 800) { validProblem = true; } } const Iz_min = Math.floor(Math.random() * 6) + 5; currentZenerState = switchClosed ? "closed" : "open"; // --- 2. MATH CALCULATIONS (FINAL) --- if (typeof currentAnswers === "undefined") currentAnswers = {}; const V_unregulated = Vin * (RL / (R1 + RL)); const isRegulating = V_unregulated > Vz; const finalVL = isRegulating ? Vz : V_unregulated; const finalIL = Math.round((finalVL / RL) * 1000); const finalIZ = totalCurrent_mA - (switchClosed ? finalIL : 0); currentAnswers.Vin = Vin; currentAnswers.Vz = Vz; currentAnswers.targetIL = finalIL; currentAnswers.targetVL = parseFloat(finalVL.toFixed(1)); currentAnswers.targetIZ = finalIZ; currentAnswers.targetPZ = Math.round(finalIZ * finalVL); // --- 3. UI SETUP --- const zenerMainGroup = document.getElementById('inputGroupZener'); if(zenerMainGroup) zenerMainGroup.style.display = "block"; const closedGroup = document.getElementById('zenerClosedQuestions'); const openGroup = document.getElementById('zenerOpenQuestions'); if (closedGroup && openGroup) { closedGroup.style.display = switchClosed ? "block" : "none"; openGroup.style.display = switchClosed ? "none" : "block"; } // --- 4. DRAWING --- ctx.strokeStyle = "black"; ctx.lineWidth = 2; ctx.fillStyle = "black"; const pink = (typeof brandPink !== 'undefined') ? brandPink : "#E64097"; // Rails drawLine(ctx, 50, 60, 480, 60); drawLine(ctx, 50, 360, 480, 360); ctx.font = "bold 16px Arial"; ctx.fillText(Vin + " V", 10, 65); ctx.fillText("0 V", 10, 365); const xReg = 180; const xLoad = 380; const yJunc = 220; // --- MAIN INPUT BRANCH --- // Input Current Value - Moved right (xReg - 65) + 800mA limit applied ctx.fillText(totalCurrent_mA + " mA", xReg - 65, 90); drawLine(ctx, xReg, 60, xReg, 110); drawArrowOnWire(ctx, xReg, 70, xReg, 90); // Series Resistor drawComponent(ctx, 'fixed', xReg, 135); drawLine(ctx, xReg, 160, xReg, 360); // Zener Diode drawHollowZenerUp(ctx, xReg, 290); ctx.fillText(Vz.toFixed(0) + " V", xReg - 65, 295); // --- RIGHT BRANCH (LOAD) --- const switchStartX = 240; const switchEndX = 290; drawLine(ctx, xReg, yJunc, switchStartX, yJunc); drawNode(ctx, xReg, yJunc); if (switchClosed) { drawLine(ctx, switchStartX, yJunc, xLoad, yJunc); ctx.beginPath(); ctx.arc(switchStartX, yJunc, 3, 0, Math.PI*2); ctx.stroke(); ctx.beginPath(); ctx.arc(switchEndX, yJunc, 3, 0, Math.PI*2); ctx.stroke(); // I_L Question ctx.strokeStyle = "black"; drawArrowOnWire(ctx, switchEndX + 15, yJunc, switchEndX + 40, yJunc); ctx.fillStyle = pink; ctx.fillText("I", switchEndX + 42, yJunc - 15); ctx.font = "10px Arial"; ctx.fillText("L", switchEndX + 47, yJunc - 10); ctx.font = "bold 16px Arial"; ctx.fillText(" = ?", switchEndX + 54, yJunc - 15); // V_L Question ctx.fillText("V", xLoad + 40, yJunc + 55); ctx.font = "10px Arial"; ctx.fillText("L", xLoad + 51, yJunc + 60); ctx.font = "bold 16px Arial"; ctx.fillText(" = ?", xLoad + 57, yJunc + 55); } else { drawLine(ctx, switchStartX, yJunc, switchEndX - 10, yJunc - 20); ctx.beginPath(); ctx.arc(switchStartX, yJunc, 3, 0, Math.PI*2); ctx.stroke(); ctx.beginPath(); ctx.arc(switchEndX, yJunc, 3, 0, Math.PI*2); ctx.stroke(); drawLine(ctx, switchEndX, yJunc, xLoad, yJunc); // I_Z Question - Moved right to xReg - 60 ctx.fillStyle = pink; drawArrowOnWire(ctx, xReg, 230, xReg, 255); ctx.fillText("I", xReg - 60, 255); ctx.font = "10px Arial"; ctx.fillText("Z", xReg - 55, 260); ctx.font = "bold 16px Arial"; ctx.fillText(" = ?", xReg - 48, 255); // P_Z Question - Moved left to xReg + 35 ctx.fillText("P", xReg + 35, 298); ctx.font = "10px Arial"; ctx.fillText("Z", xReg + 45, 303); ctx.font = "bold 16px Arial"; ctx.fillText(" = ?", xReg + 52, 298); } ctx.fillStyle = "black"; ctx.fillText("S", (switchStartX + switchEndX)/2, yJunc + 25); // --- LOAD BOX --- ctx.strokeStyle = "black"; ctx.fillStyle = "black"; ctx.font = "bold 16px Arial"; drawLine(ctx, xLoad, yJunc, xLoad, yJunc + 30); ctx.strokeRect(xLoad - 30, yJunc + 30, 60, 45); ctx.textAlign = "center"; ctx.fillText("LOAD", xLoad, yJunc + 58); ctx.textAlign = "left"; drawLine(ctx, xLoad, yJunc + 75, xLoad, 360); drawNode(ctx, xLoad, 360); // --- FOOTER --- ctx.fillStyle = "#444"; ctx.font = "italic 16px Arial"; ctx.fillText("Minimum Zener holding current = " + Iz_min + " mA", 70, 390); }
